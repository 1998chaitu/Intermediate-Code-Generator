DATA A
DATA B
DATA C
CONST E = 15
DATA D
CONST H = 25
START:
READ AX
JMP LABEL1
MOV A,AX
READ BX
LABEL2:
MOV B,BX
READ CX
MOV C,CX
LABEL1:
IF AX GT BX THEN
IF AX GT CX THEN
MOV DX, A
JMP LABEL2
ENDIF
ENDIF
IF BX GT AX THEN
IF BX GT CX THEN
MOV DX, B
ENDIF
ENDIF
IF CX GT BX THEN
IF CX GT AX THEN
MOV DX, C
ENDIF
ENDIF
MOV D, DX
PRINT D
